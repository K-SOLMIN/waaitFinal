--배송지--

CREATE TABLE ADDRESS (
	ADDR_KEY NUMBER  NOT NULL,
	CUST_KEY NUMBER  NOT NULL,
	ADDR_NAME VARCHAR2(20)  NOT NULL,
	ADDR_POSTCODE	VARCHAR2(20)  NOT NULL,
	ADDR_ADDRESS	VARCHAR2(100)  NOT NULL,
	ADDR_DETAIL VARCHAR2(100)  NOT NULL,
	ADDR_PHONE VARCHAR2(20)  NOT NULL,
	ADDR_REQUEST VARCHAR2(100)  NOT NULL,
	ADDR_DEFAULT VARCHAR2(5) CHECK (ADDR_DEFAULT IN('Y','N'))
);
